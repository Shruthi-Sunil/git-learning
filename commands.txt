git init => to initialize git repository to present workspace and it is only done once

git status => to list all untracked files

git add -A => to add all untracked files to stagingArea

git commit -m "commit message" => add from stagingArea to local repository  

git config --global user.name "username"

git config --global user.email "email"

git remote add remote_name remote_path

git push remote_name (master/main)

git branch

git checkout
